#include<stdlib.h>
struct fen
{
	char name[20];
	float s[3];
	float avr;
};


void sort(struct fen a[],int n)  
{
	int i,j;
	struct fen  t;
	for(i=1; i<n; i++)
	{
		for(j=0; j<n-i; j++)
			if(a[j+1].avr>a[j].avr) 
			{
				t=a[j];
				a[j]=a[j+1];
				a[j+1]=t;
			}
	}
}

{
	
	struct fen score[9999];
	int n,i,j,c;

		                    &score[i].s[1],&score[i].s[2]);
		score[i].avr=(score[i].s[0] +score[i].s[1] + score[i].s[2]) / 3;

	}
	c=0;
	for(j=0;j<3;j++)
	{
		for(i=0;i<n;i++)
		{
			if(score[c].s[j]<score[i].s[j])
				c=i;
			 if(i==n-1)
				printf("%c,%.2f,%.2f,%.2f\n",score[c].name,
				score[c].s[0],score[c].s[1],score[c].s[2]);
		}//输出成绩最好的
		if(j==2)
			printf("\n");
	}
	
	sort(score,n);
	for(i=0;i<n;i++)
	{
		printf("%c,%.2f,%.2f,%.2f\n",score[i].name,
				score[i].s[0],score[i].s[1],score[i].s[2]);
	    if(i==n-1)
			printf("\n");
	}//平均分
	for(i=0;i<n;i++)
